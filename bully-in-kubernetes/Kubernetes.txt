Start Docker app

Terminal:
minikube start 
minikube dashboard

2. terminal:

build/tag/push

docker build -t bully-app .

docker tag bully-app asgersong/bully-app:latest


docker push asgersong/bully-app:latest

docker pull asgersong/bully-app

3. terminal:
kubectl apply -f ./k8s/headless-service.yaml
kubectl apply -f ./k8s/deployment.yaml

4. terminal:
kubectl port-forward service/bully-service 8080:80



# Get pods info:
kubectl get pods

# Get pod IP
kubectl get pod bully-app-6656b6f6bf-5klcc -o=jsonpath='{.status.podIP}'
